export interface SynthesisOutput {
  priorities: string[];
  tasks: string[];
  themes: string[];
  blockers: string[];
  sparks: string[];       // creative ideas / open questions
  summary: string;        // brief narrative summary
}

export interface LinkerOutput {
  connections: Array<{
    currentExcerpt: string;      // text from current entry
    pastEntryId: string;         // id of connected past entry
    pastExcerpt: string;         // relevant text from past entry
    relationship: string;        // description of how they connect
  }>;
  recurringPatterns: string[];   // patterns seen across sessions
}

export interface ResearchOutput {
  references: Array<{
    title: string;
    url: string;
    snippet: string;             // relevant excerpt
    relevantTo: string;          // which spark/theme it relates to
  }>;
}

export interface ActionOutput {
  nextSteps: Array<{
    action: string;
    priority: 'high' | 'medium' | 'low';
    timeEstimate?: string;       // e.g. "30 min", "2 hours"
    context: string;             // why this action matters
  }>;
}

export interface CalendarOutput {
  proposedEvents: Array<{
    title: string;
    description: string;
    startTime: string;           // ISO 8601
    endTime: string;             // ISO 8601
    source: string;              // which action/task this came from
  }>;
}

// --- Pipeline Types ---

export type AgentName = 'synthesis' | 'linker' | 'research' | 'action' | 'calendar';

export type AgentStatus = 'idle' | 'running' | 'done' | 'error';

export interface PipelineState {
  agents: Record<AgentName, AgentStatus>;
  currentAgent: AgentName | null;
  isComplete: boolean;
  error: string | null;
}

export interface PipelineResult {
  id: string;                    // crypto.randomUUID()

  createdAt: string;             // ISO 8601
  synthesis: SynthesisOutput | null;
  linker: LinkerOutput | null;
  research: ResearchOutput | null;
  action: ActionOutput | null;
  calendar: CalendarOutput | null;
}

```typescript
import type { Entry, PipelineResult, PipelineState, AgentName } from './types';
import { getAllEntries } from './storage';
import { savePipelineResult } from './pipelineStorage';

// Agent function type — each agent is a simple async function
export type AgentFn<TInput, TOutput> = (input: TInput) => Promise<TOutput>;

// Placeholder agent functions — Plans 02/03 will provide real implementations
// The orchestrator accepts agent functions as parameters so it's testable and decoupled

interface PipelineAgents {
  synthesis: (entry: Entry) => Promise<import('./types').SynthesisOutput>;
  linker: (entry: Entry, pastEntries: Entry[], synthesisOutput: import('./types').SynthesisOutput) => Promise<import('./types').LinkerOutput>;
  research: (synthesisOutput: import('./types').SynthesisOutput) => Promise<import('./types').ResearchOutput>;
  action: (synthesisOutput: import('./types').SynthesisOutput, researchOutput: import('./types').ResearchOutput) => Promise<import('./types').ActionOutput>;
  calendar: (actionOutput: import('./types').ActionOutput) => Promise<import('./types').CalendarOutput>;
}

export async function runPipeline(
  entry: Entry,
  agents: PipelineAgents,
  onStatusChange: (state: PipelineState) => void
): Promise<PipelineResult> {
  // Implementation:
  // 1. Initialize PipelineState with all agents 'idle', currentAgent null
  // 2. For each agent in order [synthesis, linker, research, action, calendar]:
  //    a. Set agent to 'running', update currentAgent, call onStatusChange
  //    b. Call the agent function with appropriate inputs:
  //       - synthesis: receives entry
  //       - linker: receives entry, getAllEntries() (past entries), synthesis output
  //       - research: receives synthesis output
  //       - action: receives synthesis output + research output
  //       - calendar: receives action output
  //    c. On success: set agent to 'done', call onStatusChange
  //    d. On error: set agent to 'error', set state.error, call onStatusChange, STOP pipeline
  // 3. After all agents complete: set isComplete true, call onStatusChange
  // 4. Build PipelineResult with all outputs, save via savePipelineResult
  // 5. Return the PipelineResult
}
```

The orchestrator must:
- Call `getAllEntries()` to provide past entries to LinkerAgent (PERS-02)
- Filter out the current entry from past entries (don't link entry to itself)
- Handle errors gracefully — if an agent fails, mark it as error and stop
- Save the result even on partial failure (with null for failed/unrun agents)
